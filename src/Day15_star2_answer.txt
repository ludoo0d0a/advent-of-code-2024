            val mapEndIndex = input.indexOfFirst { !it.contains('#') }

        }

        private fun scaleMap(map: List<String>): MutableList<StringBuilder> {
            return map.map { row ->
                StringBuilder(row.replace("O", "[]").replace("@", "@.").flatMap { 
                    when (it) {
                        '#' -> "##"
                        else -> it.toString()
                    }
                })
            }.toMutableList()
        }

        private fun findRobot(map: List<StringBuilder>): Pair<Int, Int> {
            for (y in map.indices) {
                val x = map[y].indexOf('@')
                if (x != -1) return Pair(x, y)
            }
            throw IllegalStateException("Robot not found")
        }

        private fun moveRobot(map: MutableList<StringBuilder>, robot: Pair<Int, Int>, direction: Char) {
            val (x, y) = robot
            val (dx, dy) = when (direction) {
                '>' -> Pair(1, 0)
                '<' -> Pair(-1, 0)
                '^' -> Pair(0, -1)
                'v' -> Pair(0, 1)
                else -> throw IllegalArgumentException("Invalid direction")
            }

            var newX = x + dx
            var newY = y + dy

            if (map[newY][newX] == '[' && map[newY][newX + 1] == ']') {
                if (canPushBox(map, newX, newY, dx, dy)) {
                    pushBox(map, newX, newY, dx, dy)
                    map[y][x] = '.'
                    map[newY][newX] = '@'
                }
            } else if (map[newY][newX] == '.') {
                map[y][x] = '.'
                map[newY][newX] = '@'
            }
        }

        private fun canPushBox(map: List<StringBuilder>, x: Int, y: Int, dx: Int, dy: Int): Boolean {
            val newX = x + dx
            val newY = y + dy
            return newX in map[0].indices && newY in map.indices && map[newY][newX] == '.' && map[newY][newX + 1] == '.'
        }

        private fun pushBox(map: MutableList<StringBuilder>, x: Int, y: Int, dx: Int, dy: Int) {
            val newX = x + dx
            val newY = y + dy
            map[newY][newX] = '['
            map[newY][newX + 1] = ']'
            map[y][x] = '.'
            map[y][x + 1] = '.'
        }

        private fun displayMap(map: List<StringBuilder>) {
            println(map.joinToString("\n"))
            println()
        }

            var sum = 0L
            for (y in map.indices) {
                for (x in map[y].indices step 2) {
                    if (map[y][x] == '[' && map[y][x + 1] == ']') {
                        sum += (y + 1) * 100L + (x / 2 + 1)
                    }
                }
            }
            return sum
        }
